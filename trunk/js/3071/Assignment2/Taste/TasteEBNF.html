EBNF Grammar of Taste

Taste =      "program" ident "{" {VarDecl} {ProcDecl} {ConstDecl} "}".
VarDecl =    Type ident[ "[" number "]" ] {"," ident} ";".
ConstDecl = "const" Type Ident "=" ( "true" | "false" | number | character | "-" number ).
ProcDecl =   "void" ident "(" ")" "{" {VarDecl | Stat} "}".
Type =       "int" | "bool" | "char" .

Stat =       ident ("=" Expr [";"] | "(" ")" ";" | "[" number "]" "=" Expr ";") //"["ident"]" "=" Expr";" Does not work 
	   | "switch" "(" ident" ")" "{" "case" Expr ":" Stat [break;] {"case" Expr ":" Stat [break;] }[ "default" ":" Stat [break;] ] "}".
           | "if" "(" Expr ")" Stat ["else" Stat]
           | "while "(" Expr ")" Stat
	   | "do" Stat "(" Expr ")".
           | "read" ident ";"
           | "write" Expr ";"
	   | "for" "(" Stat Expr Stat ")" Stat.
           | "{" {Stat | VarDecl} "}".

Expr =       SimExpr [RelOp SimExpr].
SimExpr =    Term {AddOp Term}.
Term =       Factor {MulOp Factor}.
Factor =     (ident | "true" | "false" | number | character | "-" Factor).

AddOp =      "+" | "-".
MulOp =      "*" | "/".
RelOp =      "==" | "<" | ">" | "!=" | "<=" | ">=".
